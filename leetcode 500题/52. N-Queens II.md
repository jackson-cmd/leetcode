time complexity: O(n!)
space complexity: O(n^2)

```
class Solution {
    public int totalNQueens(int n) {
        return solveNQueens(n).size();
    }
    List<List<String>> ans = new ArrayList<List<String>>();
    int size;
    public List<List<String>> solveNQueens(int n) {
        this.size = n;
        char[][] empt = new char[n][n];
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j < n; ++j) {
                empt[i][j] ='.';
            }
        }
        backtrack(0, new HashSet<>(), new HashSet<>(), new HashSet<>(), empt);
        return ans;
    }
    
    public List<String> create(char[][] s) {
        List<String> b = new ArrayList();
        for(int i = 0; i < size; ++i) {
            String a = new String(s[i]);
            b.add(new String(a));
        }
        return b;
    } 
    public void backtrack(int r, Set<Integer> col, Set<Integer> diag, Set<Integer> anti, char[][] s) {
        if(r == size) {
            ans.add(create(s));
            return;
        }
        for(int c = 0; c < size; ++c) {
            int d = r-c;
            int ad = r+c;
            if(col.contains(c) || anti.contains(ad) || diag.contains(d)) continue;
            col.add(c);
            anti.add(ad);
            diag.add(d);
            s[r][c] = 'Q';
            backtrack(r+1, col, diag, anti, s);
            col.remove(c);
            anti.remove(ad);
            diag.remove(d);
            s[r][c]= '.';
        }
    }
}
```
