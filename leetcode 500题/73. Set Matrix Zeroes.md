time complexity: O(m * n)
space complexity: O(m * n)
```
class Solution {
    public void setZeroes(int[][] mat) {
        int m = mat.length, n = mat[0].length;
        int[][] tmp = new int[m][n];
        List<int[]> l = new ArrayList<>();
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(mat[i][j] == 0) l.add(new int[]{i, j});
            }
        }
        // for(int[] i:l) {
        //     System.out.println("i is " + i[0]);
        //     System.out.println("j is "+ i[1]);
        // }
        for(int[] i:l) {
            helper(mat, i[0], i[1]);
        }
    }
    public void helper(int[][] mat, int r, int c) {
        int m = mat.length, n = mat[0].length;
        for(int i = 0; i < m; ++i) mat[i][c] = 0;
        for(int j = 0; j < n; ++j) mat[r][j] = 0;
    }
}
```
