https://leetcode.com/problems/container-with-most-water/

time complexity: O(n)
space complexity: O(1)
```
class Solution {
    public int maxArea(int[] h) {
        int l = 0, r = h.length - 1;
        int ans = 0;
        while(l < r) {
            if(h[l] > h[r]) {
                ans = Math.max(ans, h[r] * (r - l));
                r--;
            } else {
                ans = Math.max(ans, h[l] * (r - l));
                l++;
            }
        }
        return ans;
    }
}
```
