https://leetcode.com/problems/letter-combinations-of-a-phone-number/


time complexity: O(4^n * n)
space complexity: O(n)
```
class Solution {
    private Map<Character, String> l = Map.of(
        '2', "abc", '3', "def", '4', "ghi", '5', "jkl", 
        '6', "mno", '7', "pqrs", '8', "tuv", '9', "wxyz");
    public List<String> letterCombinations(String dig) {
        List<String> ans = new ArrayList();
        if(dig.length() == 0) return ans;
        StringBuilder sb = new StringBuilder();
        backtrack(ans, dig, 0, sb);
        return ans;
    }
    public void backtrack(List<String> ans, String dig, int i, StringBuilder cur) {
        if(cur.length() == dig.length()) {
            ans.add(cur.toString());
            return;
        }
        String possible = l.get(dig.charAt(i));
        for(char c: possible.toCharArray()) {
            cur.append(c);
            backtrack(ans, dig, i + 1, cur);
            cur.deleteCharAt(cur.length() -1);
        }
    }
}
```
