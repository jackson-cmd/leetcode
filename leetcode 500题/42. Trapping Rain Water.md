https://leetcode.com/problems/trapping-rain-water/

time complexity: O(n)
space complexity: O(1)
```
class Solution {
    public int trap(int[] h) {
        int l = 0, r = h.length - 1;
        int max_l =0 , max_r = 0;
        int ans = 0;
        while(l < r) {
            if(h[l] < h[r]) {
                max_l = Math.max(max_l, h[l]);
                ans += max_l - h[l];
                l++;
            } else {
                max_r = Math.max(max_r, h[r]);
                ans += max_r - h[r];
                r--;
            }
        }
        return ans;
    }
}
```
