https://leetcode.com/problems/excel-sheet-column-number/

time complexity: O(n)
space complexity: O(1)
```
class Solution {
    public int titleToNumber(String c) {
        int n = c.length();
        int ans = 0;
        for(int i = n - 1; i >= 0; --i) {
            int cur = c.charAt(i) - 'A' + 1;
            ans = ans + cur * (int)Math.pow(26, (n - i - 1)); 
        }
        return ans;
    }
}
```
