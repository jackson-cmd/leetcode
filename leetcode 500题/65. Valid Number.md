time complexity: O(n)
space complexity: O(1)
有点难度
```
class Solution {
    public boolean isNumber(String s) {
        int n = s.length();
        boolean seenDot = false, seenDigit = false, seenExp = false;
        for(int i = 0; i < n; ++i) {
            char cur = s.charAt(i);
            if(Character.isDigit(cur)) {
                seenDigit = true;
            } else if(cur == '+' || cur == '-') {
                if(i > 0 && s.charAt(i-1) != 'e' && s.charAt(i-1) != 'E') return false;
            } else if(cur == 'e' || cur == 'E') {
                if(seenExp || !seenDigit) return false;
                seenExp = true;
                seenDigit = false;
            } else if(cur == '.') {
                if(seenDot || seenExp) return false;
                seenDot = true;
            } else return false;
        }
        return seenDigit;
    }
}
```
