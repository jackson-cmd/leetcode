class Solution {
    public boolean isHappy(int n) {
        // use hash set to keep track of if there is a loop
        Set<Integer> s = new HashSet();
        while(n != 1) {
            s.add(n);
            n = next(n);
            if(s.contains(n)) return false;
        }
        return true;
    }
    // find the next number
    public int next(int n) {
        int ans = 0;
        while(n > 0) {
            ans += Math.pow(n%10, 2);
            n = n/10;
        }
        return ans;
    }
}
