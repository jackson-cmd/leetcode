time complexity: O(n^2)
space complexity: O(1)
```
class Solution {
    public int[][] generateMatrix(int n) {
        int idx = 1;
        int[][] ans = new int[n][n];
        int left = 0, right = n - 1, up = 0, down = n - 1;
        while(idx <= n * n) {
            for(int i = left; i <= right; ++i) ans[up][i] = idx++;
            for(int i = up + 1; i <= down; ++i) ans[i][right] = idx++;
            if(left != right) for(int i = right - 1; i >= left; --i) ans[down][i] = idx++;
            if(up != down) for(int i = down - 1; i > up; --i) ans[i][left] = idx++;
            left++; right--;up++;down--;
        }
        return ans;
    }
}
```
