time complexity: O(n * 3^l)
space complexity: O(n)

```
class Solution {
    public boolean exist(char[][] b, String w) {
        for(int i = 0; i < b.length; ++i) {
            for(int j = 0; j < b[0].length; ++j) {
                if(backtrack(b, w, 0, i, j)) return true;
            }
        }
        return false;
    }
    public boolean backtrack(char[][] b, String w, int i, int r, int c) {
        if(i >= w.length()) return true;
        if(r < 0 || c < 0 || r == b.length || c == b[0].length || w.charAt(i) != b[r][c]) return false;
        boolean ret = false;
        b[r][c] = '!';
        int[] ro = {0, 1, 0, -1};
        int[] co = {1, 0, -1, 0};
        for(int id = 0; id < 4; ++id) {
            ret = backtrack(b, w, i + 1, r + ro[id], c + co[id]);
            if(ret) break;
        }
        b[r][c] = w.charAt(i);
        return ret;
    }
}
```
