time complexity: O(n)
space complexity: O(1)
```
class Solution {
    public void reverseWords(char[] s) {
        reverse(s, 0, s.length - 1);
        reverseWord(s);
    } 
    public void reverse(char[] s, int l, int r) {
        while (l < r) {
            char tmp = s[l];
            s[l++] = s[r];
            s[r--] = tmp;
        }
    }
    public void reverseWord(char[] s) {
        int n = s.length;
        int l = 0, r = 0;
        while (r < n) {
            while(r< n && s[r] != ' ') r++;
            reverse(s, l, r-1);
            l = r+1;
            r = r+1;
        }
    }
}
```
