time complexity: O(n)
space complexity: O(1)
```
class Solution {
    public void rotate(int[][] m) {
        //first reverse matrix symeetry, then reverse left to right
        int n = m.length;
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j <= i; ++j) {
                if(i == j) continue;
                int tmp = m[i][j];
                m[i][j] = m[j][i];
                m[j][i] = tmp;
            }
        }
        int i = 0, j = n-1;
        while(i < j) {
            for(int a = 0; a < n; ++a) {
                int tmp = m[a][i];
                m[a][i] = m[a][j];
                m[a][j] = tmp;
            }
            i++;
            j--;
        }
    }
}
```
