time complexity: O(n)
space complexity: O(1)

```
class Solution {
    public int myAtoi(String s) {
        s = s.strip();
        if(s.length() == 0) return 0;
        int sign = 1;
        int ans = 0;
        int Max = Integer.MAX_VALUE, Min = Integer.MIN_VALUE;
        char first = s.charAt(0);
        if(first == '-') sign = -1;
        else if(first == '+') sign = 1;
        else {
            if (Character.isDigit(first)) ans += first - '0';
            else return 0;
        }
        for(int i = 1; i < s.length(); ++i) {
            char cur = s.charAt(i);
            if(!Character.isDigit(cur)) return sign * ans;
            else {
                int digit = cur - '0';
                if((ans > Max / 10) || (ans == Max/10 && Max - ans * 10 < digit)){
                    return sign == -1? Min: Max;
                }
                ans = ans * 10 + digit;
            }
        }
        return sign * ans;
    }
}
```
