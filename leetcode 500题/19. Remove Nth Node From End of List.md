time complexity: O(n)
space complexity: O(1)
```
class Solution {
    public ListNode removeNthFromEnd(ListNode head, int n) {
        //use slow and faster pointers
        //we first iterate n, and use similar idea like sliding windows, and when the faster pointer touch the null, then slower pointers pointer to our deleted node
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode fast = dummy;
        int i = 0;
        while(i < n && fast.next != null) {
            fast = fast.next;
            i++;
        }
        //creat our slow pointer
        ListNode slow = dummy;
        while(fast.next != null && fast!= null) {
            slow = slow.next;
            fast = fast.next;
        }
        slow.next = slow.next.next;
        return dummy.next;
    }
}
```
