time complexity: O(logn)
space complexity: O(1)
```
class Solution {
    public int mySqrt(int x) {
        if(x == 0) return 0;
        if(x < 4) return 1; 
        int l = 0, r = x/2;
        long num;
        while (l <= r) {
            int mid = l + (r - l)/2;
            num = (long) mid * mid;
            if(num < x) l = mid + 1;
            else if(num > x) r = mid - 1;
            else return mid;
        }
        return r;
    }
}
```
