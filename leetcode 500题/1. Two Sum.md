time complexity: O(n)
space complexity: O(1)
并不明白这个解法，但是确实beat 100%
```
class Solution {
    
    public int[] twoSum(int[] nums, int target) {
        for (int l = 1; l <= nums.length - 1; l++ ) {
            for ( int k = 0; k < nums.length -l; k++  ) {
                if (nums[k] + nums[k+l] == target) {
                    return new int[]{k,k+l };
                }
            }
        }
        return null;
    }
}
```
