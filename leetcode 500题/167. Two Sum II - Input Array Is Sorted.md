time complexity: O(n)
space complexity: O(1)
```

class Solution {
    public int[] twoSum(int[] nums, int t) {
        int l = 0, n = nums.length;
        int r = n -1;
        while(l < r) {
            if(nums[l] + nums[r] < t) {
                l++;
            } else if(nums[l] + nums[r] > t) {
                r--;
            } else {
                return new int[]{l + 1, r + 1};
            }
        }
        return new int[]{-1, -1};
    }
}
```
