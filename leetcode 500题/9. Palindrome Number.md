https://leetcode.com/problems/palindrome-number/

time complexity: O(log(n))
space complexity: O(1)
```
class Solution {
    public boolean isPalindrome(int x) {
        if (x < 0) return false;
        if(x == 0) return true;
        int ans = 0;
        int tmp = x;
        while(x != 0) {
            int mod = x%10;
            ans = ans * 10 + mod;
            x /= 10;
        }
        return tmp - ans == 0; 
    }
}
```
