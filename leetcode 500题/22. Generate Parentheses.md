time complexity: O(4^n/(n^1/2))
space complexity: O(4^n/(n^1/2))
```
class Solution {
    public List<String> generateParenthesis(int n) {
        StringBuilder sb = new StringBuilder();
        ArrayList<String> ans = new ArrayList();
        helper(0, 0, sb, ans, n);
        return ans;
    }
    public void helper(int open, int close, StringBuilder sb, ArrayList<String> ans, int max) {
        if(sb.length() == max * 2) {
            ans.add(sb.toString());
            return;
        }
        if(open < max) {
            sb.append("(");
            helper(open + 1, close, sb, ans, max);
            sb.deleteCharAt(sb.length() - 1);
        } 
        if(open > close) {
            sb.append(")");
            helper(open, close + 1, sb, ans, max);
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}
```
