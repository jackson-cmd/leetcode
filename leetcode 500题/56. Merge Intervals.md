https://leetcode.com/problems/merge-intervals/

time complexity: O(nlogn)
space complexity: O(n)
```
class Solution {
    public int[][] merge(int[][] in) {
        //sort the array and use arraylist
        Arrays.sort(in, (a, b) -> a[0] - b[0]);
        List<int[]> l = new ArrayList();
        for(int[] i: in) {
            if(l.size() == 0) l.add(in[0]);
            else {
                int[] pre = l.get(l.size()-1);
                if(i[0] <= pre[1]) l.get(l.size()-1)[1] = Math.max(i[1], pre[1]);
                else l.add(i);
            }
        }
        //return l.toArray(new int[l.size()][2]);
        int[][] ans = new int[l.size()][2];
        for(int i = 0; i < l.size();++i) {
            ans[i][0] = l.get(i)[0];
            ans[i][1] = l.get(i)[1];
        }
        return ans;
    }
}
```
