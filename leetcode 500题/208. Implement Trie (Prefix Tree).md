class Trie {
    class TrieNode {
        public boolean isWord;
        public TrieNode[] child;
        public TrieNode() {
            isWord = false;
            child = new TrieNode[26];
        }
    }
    TrieNode root;
    public Trie() {
        root = new TrieNode();
    }
    
    public void insert(String w) {
        TrieNode p = root;
        for(int i = 0; i < w.length(); ++i) {
            int idx = (int)(w.charAt(i) - 'a');
            if(p.child[idx] == null) p.child[idx] = new TrieNode();
            p = p.child[idx];
        }
        p.isWord = true;
    }
    //find the last char
    public TrieNode find(String w) {
        TrieNode p = root;
        for(int i = 0; i < w.length(); ++i) {
            int idx = (int)(w.charAt(i) - 'a');
            if(p.child[idx] == null) return null;
            p = p.child[idx];
        }
        return p;
        
    }
    public boolean search(String w) {
        TrieNode cur = find(w);
        return  cur != null && cur.isWord;
    }
    
    public boolean startsWith(String pre) {
        TrieNode cur = find(pre);
        return  cur != null;
    }
}
